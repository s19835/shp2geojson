──── Available Commands ─────────────────────────────────────────────
/status Print live worker throughput and queue depth
/pause Gracefully pause all workers, write checkpoint
/resume Resume paused workers
/workers <N> Dynamically scale worker count up or down
/skip <file> Skip a specific pending file, mark as SKIPPED
/log Tail conversion_errors.log inline
/dry-run Preview remaining pending files without converting
/quit Checkpoint state and exit cleanly
Example: /workers 16 → scales from 8 to 16 workers immediately Example: /skip broken/file.shp → removes from queue, logs as SKIPPED
Ctrl+C — pause and checkpoint · Ctrl+D — force quit (no checkpoint)
.shp2geojson.toml
# .shp2geojson.toml — version-controllable project config
[conversion] reproject = true # auto-reproject to WGS84 output_format = "geojson" # or "geojsonl" overwrite = false jobs = 0 # 0 = auto (logical CPUs)
[output] mirror_structure = true log_file = "./logs/conversion_errors.log"
[hooks] # Template vars: {{file}} {{output}} {{features}} {{duration_ms}} {{reason}} {{summary_json}}
on_file_complete = "aws s3 cp {{output}} s3://my-bucket/geojson/ --quiet"
on_file_failed = "curl -s https://hooks.slack.com/T.../B... \ -d '{\"text\": \"❌ Failed: {{file}} — {{reason}}\"}'
on_batch_done = "python3 ingest.py --summary '{{summary_json}}'"
